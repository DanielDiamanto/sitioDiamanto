CREATE TABLE IF NOT EXISTS `Personas` ( `idPersona` int NOT NULL AUTO_INCREMENT, `Nombre` varchar(50) NOT NULL, `Apellido_Paterno` varchar(50) NOT NULL, `Apellido_Materno` varchar(50) NOT NULL, `Edad` varchar(6) NOT NULL, `Estado_Civil` varchar(100) NOT NULL, `Municipio` varchar(250) CHARACTER SET utf8 NOT NULL, `E-mail` varchar(50) NOT NULL, PRIMARY KEY (`idPersona`) ) ENGINE=MyISAM DEFAULT CHARSET=latin1

CREATE TABLE IF NOT EXISTS `Documentos` ( `idDoc` int NOT NULL AUTO_INCREMENT, `Nombre` varchar(50) NOT NULL, `Relevancia` varchar(50) NOT NULL, `PDF` varchar(50) NOT NULL, PRIMARY KEY (`idDoc`) ) ENGINE=MyISAM DEFAULT CHARSET=latin1
